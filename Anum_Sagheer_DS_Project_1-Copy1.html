from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.edge.service import Service
from bs4 import BeautifulSoup
import pandas as pd
import time

service = Service('C:/Users/user/Downloads/msedgedriver.exe')  
driver = webdriver.Edge(service=service)

try:

    url = "https://www.spaceweatherlive.com/en/solar-activity/top-50-solar-flares"
    driver.get(url)
    time.sleep(5)  

    # Parse the HTML content
    soup = BeautifulSoup(driver.page_source, 'html.parser')
    print(soup.prettify())
    table = soup.find('table')
    rows = table.find_all('tr')

    # Extract data from the table
    data = []
    for row in rows[1:]:  # Skip the header row
        cols = row.find_all('td')
        data.append([col.text.strip() for col in cols])

    # Create DataFrame with appropriate column names
    df_solar = pd.DataFrame(data, columns=['rank', 'x_class', 'date', 'region', 'start_time', 'max_time', 'end_time', 'movie'])
    print("SpaceWeatherLive Data:")
    print(df_solar.head(50))  

finally:
    driver.quit()

    lines = nasa_table.text.splitlines()
    nasa_data = [line.split() for line in lines[13:] if line.strip()]  # Skip header lines

    # Find the row with the maximum length to identify how many columns are actually needed
    max_columns = max(len(row) for row in nasa_data)

    # Adjust data rows to match the detected maximum length
    # Pad shorter rows with None values to ensure consistent row length
    nasa_data_aligned = [row + [None] * (max_columns - len(row)) for row in nasa_data]

    # Define appropriate column names based on your expected column structure
    columns = ['start_date', 'start_time', 'end_date', 'end_time', 'start_frequency', 'end_frequency',

# Function to find the best matching row in the NASA data for each SpaceWeatherLive flare
def find_best_match(row, nasa_df):
    # Filter potential matches by classification letter
    potential_matches = nasa_df[nasa_df['flare_letter'] == row['flare_letter']]
    
    # If exact classification number matches are available, use them
    exact_matches = potential_matches[np.isclose(potential_matches['flare_number'], row['flare_number'], atol=0.1)]
    if not exact_matches.empty:
        exact_matches['date_diff'] = (exact_matches['start_datetime'] - row['date']).abs()
        return exact_matches.loc[exact_matches['date_diff'].idxmin()]  # Closest date match among exact matches
    
    # If no exact matches, find the closest classification by number
    potential_matches['number_diff'] = (potential_matches['flare_number'] - row['flare_number']).abs()
    closest_classification = potential_matches.loc[potential_matches['number_diff'].idxmin()]
    
    # Calculate the absolute date difference directly for the closest classification
    date_diff = abs(closest_classification['start_datetime'] - row['date'])
    closest_classification['date_diff'] = date_diff
    return closest_classification
